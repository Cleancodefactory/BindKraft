<message> :== <any>;
<response> :==
{
	status: {
		issuccessful: <bool>,
		message: <string?>,								// Autogenerated message should be supplied when nothing can be extracted from the actual response
		?title: <string?>,
		?returnurl: <string?>,
		?isprobing: <bool?>,								// The response is a result of browser autgenerated request (some browsers, under certain confgiruation settings can make your life hard)
		?isreadonly: <bool?>								// Legacy property with very unclear meaning
		?messages: <array?(<message>)>
	},
	?views: <object?(<string>,<string>)>,
	?data: <any>,
	?metadata: <any>,
	?resources: <object?(<string>,<object(<string>,<string>)>)>,
	?lookups: <object?(<string>,<array?(<object>)>)>
};
